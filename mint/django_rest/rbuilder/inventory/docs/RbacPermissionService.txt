RbacPermission properties:
    created_by - user who created this resource
    created_date - creation date
    grant_id - the database ID for the permission
    modified_by - user who last modified this resource
    modified_date - modification date
    permission - permission
    queryset - queryset id
    role - rbac_role id

Methods:
    GET:
        Authentication: admin
    POST:
        not supported
    PUT:
        Authentication: admin
    DELETE:
        Authentication: admin
