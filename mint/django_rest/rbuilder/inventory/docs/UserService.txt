User properties:
    can_create - User can create resources?
    created_by - User created by
    created_date - User created date
    display_email - User display email
    email - User email
    full_name - User full name
    last_login_date - User last login date
    modified_by - User modified by
    modified_date - User modified date
    user_id - User id
    user_name - User name

Methods:
    GET:
        Authentication: user
    POST:
        not supported
    PUT:
        Authentication: user
    DELETE:
        Authentication: admin
